# 1 安装jdk #

比较常见,我这里以1.8为例

## 1.1 解压复制 ##

    sudo tar -zxvf /home/xjy/Downloads/jdk-8u144-linux-x64.tar.gz -C /usr/lib/jvm

## 1.2 执行安装 ##

    sudo update-alternatives --install "/usr/bin/javac" "javac" "/usr/lib/jvm/jdk1.8.0_144/bin/javac" 1 \  &amp;&amp; sudo update-alternatives --set javac /usr/lib/jvm/jdk1.8.0_144/bin/javac \ &amp;&amp; sudo update-alternatives --install "/usr/bin/java" "java" "/usr/lib/jvm/jdk1.8.0_144/bin/java" 1 \ &amp;&amp; sudo update-alternatives --set java /usr/lib/jvm/jdk1.8.0_144/bin/java

## 1.3 验证版本 ##  

    java -version

## 1.4 另外可增加配置 ##

    export JAVA_HOME=/usr/lib/jvm/jdk1.8.0_144
    export PATH=$JAVA_HOME/bin:$PATH
    export CLASSPATH=.:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar

# 2 安装maven #

## 2.1 下载安装包 ##

    wget http://mirrors.cnnic.cn/apache/maven/maven-3/3.0.5/binaries/apache-maven-3.0.5-bin.tar.gz

## 2.2 解压 ##

    tar zvxf apache-maven-3.0.5-bin.tar.gz

## 2.3 移到指定目录 ##

    mv apache-maven-3.0.5 /usr/local

## 2.4 编辑配置文件/etc/profile，加入 ##

    export MAVEN_HOME=/usr/local/apache-maven-3.0.5
    export PATH=$PATH:$MAVEN_HOME/bin

## 2.5 始配置生效 ##

    source /etc/profile

## 2.6 确认安装完成 ##

    mvn -v

# 3 nexus安装 #

## 3.1 下载安装包 ##

    wget https://sonatype-download.global.ssl.fastly.net/nexus/3/nexus-3.5.2-01-unix.tar.gz

## 3.2 解压并复制 ##

    sudo tar -zxvf nexus-3.5.2-01-unix.tar.gz -C /usr/lib/nexus-xjy

## 3.3 做链接 ##

    sudo ln -s /usr/lib/nexus-xjy/nexus-3.5.2-01/bin/nexus /etc/init.d/nexus

## 3.4 赋权限 ##

    chmod 755 /etc/init.d/nexus

## 3.5 设置开机启动并运行 ##

    cd /etc/init.d
    chkconfig --add nexus
    chkconfig --levels 345 nexus on
    service nexus start
    tail -fn 200 /usr/lib/nexus-xjy/sonatype-work/nexus3/log

# 4 svn安装 #

## 4.1 yum安装svnversion ##

4.1.1 检查是否安装了低版本的SVN

    rpm -qa subversion

4.1.2 如果存储旧版本，卸载旧版本SVN

    yum remove subversion

4.1.3 安装SVN

    yum install subversion

4.1.4 验证安装版本信息

    svnserve --version

4.1.5 代码库创建

    mkdir -p /opt/svn/
    svnadmin create /opt/svn/repo

## 4.2 账户权限及防火墙配置 ##

4.2.1  编辑/opt/svn/repo/conf/passwd，加入如下内容：

<pre>
[users]
# harry = harryssecret
# sally = sallyssecret
xjy= 123
www = 123
</pre>

4.2.2  编辑/opt/svn/repo/conf/authz，加入如下内容：

<pre>
[groups]
#创建一个admin组，将用户加入到组
admin = xjy,www
[/] #根目录权限设置（就是“svn”这个文件夹）
xjy = rw            #xjy对svn下的所有版本库有读写权限
www = r             #www对svn下的所有版本库有读权限
[repo:/]            #repo:/,表示对repo版本库下的所有资源设置权限
@admin = rw         #admin组的用户对repo版本库有读写权限
</pre>

4.2.3  编辑/opt/svn/repo/conf/svnconfig.conf，加入如下内容：

<pre>
[general]  
#匿名访问的权限，可以是read,write,none,默认为read  
anon-access = none  
#使授权用户有写权限  
auth-access = write  
#密码数据库的路径  
password-db = passwd  
#访问控制文件  
authz-db = authz  
#认证命名空间，subversion会在认证提示里显示，并且作为凭证缓存的关键字  
realm = /opt/svn/repo
</pre>

4.2.4  编辑/etc/sysconfig/iptables，加入如下内容

<pre>
-A INPUT -m state --state NEW -m tcp -p tcp --dport 3690 -j ACCEPT
</pre>

## 4.3 启动SVN ##

4.3.1 启动SVN

    svnserve -d -r /opt/svn/

4.3.2 查看SVN进程

    ps -ef|grep svnserve

4.3.3 检测SVN端口

    netstat -ln |grep 3690

## 4.4 完成上面的步骤即可在客户端正常访问

地址：[svn://ip地址/repo](svn://ip地址/repo)

# 5 jenkins安装 #

## 5.1 安装 ##

    sudo wget -O /etc/yum.repos.d/jenkins.repo https://pkg.jenkins.io/redhat-stable/jenkins.repo
    sudo rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io.key
    yum install jenkins

## 5.2 启动并加入开机启动项 ##

    sudo service jenkins start
    sudo chkconfig jenkins on

## 5.3 访问jenkins的web管理端 ##

[http://ip:8080](http://ip地址:8080)
